team_dic  = {'삼':['삼전','교대근무','연봉 많이 성과금 듬뿍'], 
             '쿠':['쿠팡','적절한 외근','연봉 full'],
             '카':['카카오','인턴말고 정규직','사내규정'], 
             '구':['구글','정규직이지만 자유로운','사장보다 연봉 높은 막내'], 
             '네':['네이버','워라벨','많이']}
name = input()
import random
list0 = team_dic[name]
random.choice(list0)
print((f'우리 {name}님은 {list0[0]}를 목표로 하고 있습니다.'))
print((f'{list0[1]}근무 형태를 희망합니다.'))
print((f'{name}님은 {list0[2]} 받을 만한 분이십니다.'))
